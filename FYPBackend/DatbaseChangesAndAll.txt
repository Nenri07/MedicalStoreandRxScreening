-- Table 1: subscriptions
CREATE TABLE subscriptions (
    id               INT IDENTITY(1,1) PRIMARY KEY,
    cust_id          INT NOT NULL,
    store_id         INT NOT NULL,
    day_of_month     INT NOT NULL,        -- 1 to 28, day each month order fires
    next_order_date  DATE NOT NULL,       -- when next auto order will be placed
    status           VARCHAR(20) NOT NULL DEFAULT 'active',  -- active/paused/cancelled
    created_at       DATETIME DEFAULT GETDATE(),

    FOREIGN KEY (cust_id)  REFERENCES customer(c_id),
    FOREIGN KEY (store_id) REFERENCES medicalstore(store_id)
);

-- Table 2: subscription_medicines (what medicines are in each subscription)
CREATE TABLE subscription_medicines (
    id                INT IDENTITY(1,1) PRIMARY KEY,
    subscription_id   INT NOT NULL,
    base_name         VARCHAR(100) NOT NULL,
    quantity          INT NOT NULL,

    FOREIGN KEY (subscription_id) REFERENCES subscriptions(id)
);
